41int interpolate_grey(int p11, int p21, int p12, int p22, float dx, float dy)
42{
43 float interpolated_value =
44 p11 * (1 - dx) * (1 - dy) +
45 p21 * dx * (1 - dy) +
46 p12 * (1 - dx) * dy +

186void areaScrambling_PGM(ImageBase &image, ImageBase &o_image, int start_i, int start_j, int area_h, int area_w)
187{
188 int nbPixels = area_h * area_w;
189 std::random_device randev;
190 std::mt19937 rng(randev());
191 std::uniform_int_distribution<unsigned int> distrbit(0, nbPixels - 1);
192 ImageBase unavailability(area_w, area_h, false);
193 for (int i = 0; i < area_h; ++i)
194 for (int j = 0; j < area_w; ++j)
195 {
196 int randIdx = distrbit(rng);
197 int randJ = randIdx % area_w;
198 int randI = (randIdx - randJ) / area_w;
199 while (unavailability[randI][randJ] == 1)
200 {
201 randIdx = (randIdx == nbPixels - 1 ? 0 : randIdx + 1);
202 randJ = randIdx % area_w;
203 randI = (randIdx - randJ) / area_w;
204 }
205 unavailability[randI][randJ] = 1;
206 o_image[start_i + randI][start_j + randJ] = image[start_i + i][start_j + j];
207 }
208}

320 unsigned int bSeq[8];
